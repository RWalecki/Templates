\newcommand{\Figure}[2]%{{{
{
    \begin{figure}[h!]
        \begin{center}
            \includegraphics[width=\textwidth,height=0.65\textheight,keepaspectratio]{#1}
            \caption{#2}
        \end{center}
    \end{figure}
}%}}}

\newcommand{\FrameFull}[3]%{{{
{
\begin{frame}
\frametitle{#1}
\framesubtitle{#2}
#3
\end{frame}
}%}}}

\newcommand{\FrameSplit}[4]%{{{
{
\FrameFull{#1}{#2}
    {
        \begin{columns}
            \begin{column}{.3\linewidth}
                #3
        \end{column}
            \begin{column}{.7\linewidth}
                #4
            \end{column}
        \end{columns}
    }
}%}}}
